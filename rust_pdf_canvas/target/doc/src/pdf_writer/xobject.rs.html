4/// Writer for an _image XObject stream_.
5///
6/// This struct is created by [`Chunk::image_xobject`].
7pub struct ImageXObject<'a> {
8    stream: Stream<'a>,
9}
10
11impl<'a> ImageXObject<'a> {
12    /// Create a new image stream writer.
13    pub(crate) fn start(mut stream: Stream<'a>) -> Self {
14        stream.pair(Name(b"Type"), Name(b"XObject"));
15        stream.pair(Name(b"Subtype"), Name(b"Image"));
16        Self { stream }
17    }
18
19    /// Write the `/Width` attribute.
20    pub fn width(&mut self, width: i32) -> &mut Self {
21        self.pair(Name(b"Width"), width);
22        self
23    }
24
25    /// Write the `/Height` attribute.
26    pub fn height(&mut self, height: i32) -> &mut Self {
27        self.pair(Name(b"Height"), height);
28        self
29    }
30
31    /// Start writing the `/ColorSpace` attribute.
32    ///
33    /// Required for all images except if using the `JPXDecode` filter.
34    /// If this is an image soft mask, the color space must be `DeviceGray`.
35    /// Must not be `Pattern`.
36    pub fn color_space(&mut self) -> ColorSpace<'_> {
37        self.insert(Name(b"ColorSpace")).start()
38    }
39
40    /// Write the `/ColorSpace` attribute as a name from the resource dictionary.
41    ///
42    /// Required for all images except if using the `JPXDecode` filter.
43    /// If this is an image soft mask, the color space must be `DeviceGray`.
44    /// Must not be `Pattern`.
45    pub fn color_space_name(&mut self, name: Name) -> &mut Self {
46        self.pair(Name(b"ColorSpace"), name);
47        self
48    }
49
50    /// Write the `/BitsPerComponent` attribute. Required.
51    ///
52    /// Required for all images except if using the `JPXDecode` filter.
53    pub fn bits_per_component(&mut self, bits: i32) -> &mut Self {
54        self.pair(Name(b"BitsPerComponent"), bits);
55        self
56    }
57
58    /// Write the `/Intent` attribute. PDF 1.1+.
59    pub fn intent(&mut self, intent: RenderingIntent) -> &mut Self {
60        self.pair(Name(b"Intent"), intent.to_name());
61        self
62    }
63
64    /// Write the `/ImageMask` attribute to set whether this image is a clipping
65    /// mask. If so, the `/BitsPerComponent` must be `1` and `/Mask` and
66    /// `/ColorSpace` attributes shall be left undefined.
67    pub fn image_mask(&mut self, mask: bool) -> &mut Self {
68        self.pair(Name(b"ImageMask"), mask);
69        self
70    }
71
72    /// Write the `/Mask` attribute to set a color key mask. The iterable color
73    /// argument must contain a range of colors (minimum and maximum) for each
74    /// channel that shall be masked out. PDF 1.3+.
75    pub fn color_mask(&mut self, colors: impl IntoIterator<Item = i32>) -> &mut Self {
76        self.insert(Name(b"Mask")).array().typed().items(colors);
77        self
78    }
79
80    /// Write the `/Mask` attribute to set another image as the stencil mask of
81    /// this image.
82    pub fn stencil_mask(&mut self, mask: Ref) -> &mut Self {
83        self.pair(Name(b"Mask"), mask);
84        self
85    }
86
87    /// Write the `/Decode` attribute to set the decoding of the image sample
88    /// colors to the specified color space. Must have twice the amount of
89    /// elements as the color space.
90    pub fn decode(&mut self, decode: impl IntoIterator<Item = f32>) -> &mut Self {
91        self.insert(Name(b"Decode")).array().typed().items(decode);
92        self
93    }
94
95    /// Write the `/Interpolate` attribute.
96    pub fn interpolate(&mut self, interpolate: bool) -> &mut Self {
97        self.pair(Name(b"Interpolate"), interpolate);
98        self
99    }
100
101    /// Write the `/Alternates` attribute. PDF 1.3+.
102    ///
103    /// Images that may replace this image. The order is not relevant.
104    pub fn alternates(&mut self, alternates: impl IntoIterator<Item = Ref>) -> &mut Self {
105        self.insert(Name(b"Alternates")).array().items(alternates);
106        self
107    }
108
109    /// Start writing the `/SMask` attribute. PDF 1.4+.
110    ///
111    /// Must not be used if this image already is an image soft mask.
112    pub fn s_mask(&mut self, x_object: Ref) -> &mut Self {
113        self.pair(Name(b"SMask"), x_object);
114        self
115    }
116
117    /// Write the `/SMaskInData` attribute. PDF 1.5+.
118    ///
119    /// May only be used for images that use the `JPXDecode` filter. If set to
120    /// something other than `Ignore`, the `SMask` attribute must not be used.
121    pub fn s_mask_in_data(&mut self, mode: SMaskInData) -> &mut Self {
122        self.pair(Name(b"SMaskInData"), mode.to_int());

124    }
125
126    /// Write the `/StructParent` attribute to indicate the [structure tree

131    }

133    /// Write the `/Matte` attribute for image soft masks. PDF 1.4+.
134    ///
135    /// This shall be the matte color of the parent image encoded in its color
136    /// space.
137    pub fn matte(&mut self, color: impl IntoIterator<Item = f32>) -> &mut Self {
138        self.insert(Name(b"Matte")).array().items(color);
139        self
140    }

142    /// Write the `/Metadata` attribute to specify the image's metadata. PDF
143    /// 1.4+.
144    ///

146    pub fn metadata(&mut self, id: Ref) -> &mut Self {

148        self
149    }

154/// What to do with in-data mask information in `JPXDecode` images.
155pub enum SMaskInData {
156    /// Discard the mask data.
157    Ignore,
158    /// Use the mask data.
159    Use,
160    /// Use the mask data on the image whose backdrop has been pre-blended with
161    /// a matte color.
162    Preblended,
163}
164
165impl SMaskInData {
166    pub(crate) fn to_int(&self) -> i32 {
167        match self {
168            Self::Ignore => 0,
169            Self::Use => 1,
170            Self::Preblended => 2,
171        }
172    }
173}

175/// Writer for an _form XObject stream_. PDF 1.1+.

260/// Writer for a _group XObject dictionary_. PDF 1.4+.
261///
262/// This struct is created by [`FormXObject::group`] and [`Page::group`].
263pub struct Group<'a> {
264    dict: Dict<'a>,
265}
266
267writer!(Group: |obj| {
268    let mut dict = obj.dict();
269    dict.pair(Name(b"Type"), Name(b"Group"));
270    Self { dict }
271});
272
273impl<'a> Group<'a> {
274    /// Set the `/S` attribute to `/Transparency`. Required to set the remaining
275    /// transparency parameters.
276    pub fn transparency(&mut self) -> &mut Self {
277        self.pair(Name(b"S"), Name(b"Transparency"));
278        self
279    }
280
281    /// Start writing the `/CS` attribute to set the color space.
282    ///
283    /// This is optional for isolated groups and required for groups where the
284    /// color space cannot be derived from the parent.
285    pub fn color_space(&mut self) -> ColorSpace<'_> {
286        self.insert(Name(b"CS")).start()
287    }
288
289    /// Set the `/I` attribute to indicate whether the group is isolated.
290    ///
291    /// If it is true, the group will initially be composited against a clear
292    /// backdrop. If it is false, the group will be composited against the
293    /// backdrop of the parent group.
294    pub fn isolated(&mut self, isolated: bool) -> &mut Self {
295        self.pair(Name(b"I"), isolated);
296        self
297    }
298
299    /// Set the `/K` attribute to indicate whether the group is a knockout
300    /// group.
301    ///
302    /// Within a knockout group, the group children are all composited
303    /// separately against the backdrop instead of on top of each other.
304    pub fn knockout(&mut self, knockout: bool) -> &mut Self {
305        self.pair(Name(b"K"), knockout);
306        self
307    }
308}
