3/// Writer for an _annotation dictionary_.
4///
5/// An array of this struct is created by [`Page::annotations`].
6pub struct Annotation<'a> {
7    dict: Dict<'a>,
8}
9
10writer!(Annotation: |obj| {
11    let mut dict = obj.dict();
12    dict.pair(Name(b"Type"), Name(b"Annot"));
13    Self { dict }
14});
15
16impl<'a> Annotation<'a> {
17    /// Write the `/Subtype` attribute to tell the viewer the type of this
18    /// particular annotation.
19    pub fn subtype(&mut self, kind: AnnotationType) -> &mut Self {
20        self.pair(Name(b"Subtype"), kind.to_name());
21        self
22    }
23
24    /// Write the `/Rect` attribute. This is the location and dimensions of the
25    /// annotation on the page.
26    pub fn rect(&mut self, rect: Rect) -> &mut Self {
27        self.pair(Name(b"Rect"), rect);
28        self
29    }
30
31    /// Write the `/Contents` attribute. This is the content or alt-text,
32    /// depending on the [`AnnotationType`].
33    pub fn contents(&mut self, text: TextStr) -> &mut Self {
34        self.pair(Name(b"Contents"), text);
35        self
36    }
37
38    /// Write the `/NM` attribute. This uniquely identifies the annotation on the
39    /// page. PDF 1.3+.
40    pub fn name(&mut self, text: TextStr) -> &mut Self {
41        self.pair(Name(b"NM"), text);
42        self
43    }
44
45    /// Write the `/M` attribute, specifying the date the annotation was last
46    /// modified. PDF 1.1+.
47    pub fn modified(&mut self, date: Date) -> &mut Self {
48        self.pair(Name(b"M"), date);
49        self
50    }
51
52    /// Write the `/F` attribute.
53    pub fn flags(&mut self, flags: AnnotationFlags) -> &mut Self {
54        self.pair(Name(b"F"), flags.bits() as i32);
55        self
56    }
57
58    /// Write the `/Border` attribute. This describes the look of the border
59    /// around the annotation, including width and horizontal and vertical
60    /// border radii. The function may also receive a dash pattern which
61    /// specifies the lengths and gaps of the border segments on a dashed
62    /// border. Although all PDF versions accept `/Border`, this feature
63    /// specifically is only available in PDF 1.1 or later.
64    pub fn border(
65        &mut self,
66        h_radius: f32,
67        v_radius: f32,
68        width: f32,
69        dash_pattern: Option<&[f32]>,
70    ) -> &mut Self {
71        let mut array = self.insert(Name(b"Border")).array();
72        array.item(h_radius);
73        array.item(v_radius);
74        array.item(width);
75
76        if let Some(pattern) = dash_pattern {
77            array.push().array().items(pattern);
78        }
79
80        array.finish();
81        self
82    }
83
84    /// Start writing the `/BS` attribute. These are some more elaborate border
85    /// settings taking precedence over `/B` for some annotation types. PDF 1.2+.
86    pub fn border_style(&mut self) -> BorderStyle<'_> {
87        self.insert(Name(b"BS")).start()
88    }
89
90    /// Write the `/C` attribute forcing a transparent color. This sets the
91    /// annotations background color and its popup title bar color. PDF 1.1+.
92    pub fn color_transparent(&mut self) -> &mut Self {
93        self.insert(Name(b"C")).array();
94        self
95    }
96
97    /// Write the `/C` attribute using a grayscale color. This sets the
98    /// annotations background color and its popup title bar color. PDF 1.1+.
99    pub fn color_gray(&mut self, gray: f32) -> &mut Self {
100        self.insert(Name(b"C")).array().item(gray);
101        self
102    }
103
104    /// Write the `/C` attribute using an RGB color. This sets the annotations
105    /// background color and its popup title bar color. PDF 1.1+.
106    pub fn color_rgb(&mut self, r: f32, g: f32, b: f32) -> &mut Self {
107        self.insert(Name(b"C")).array().items([r, g, b]);
108        self
109    }
110
111    /// Write the `/C` attribute using a CMYK color. This sets the annotations
112    /// background color and its popup title bar color. PDF 1.1+.
113    pub fn color_cmyk(&mut self, c: f32, m: f32, y: f32, k: f32) -> &mut Self {
114        self.insert(Name(b"C")).array().items([c, m, y, k]);
115        self
116    }
117
118    /// Write the `/StructParent` attribute to indicate the [structure tree
119    /// element][StructElement] this annotation belongs to. PDF 1.3+.
120    pub fn struct_parent(&mut self, key: i32) -> &mut Self {
121        self.pair(Name(b"StructParent"), key);
122        self
123    }
124
125    /// Start writing the `/A` dictionary. Only permissible for the subtype
126    /// `Link`.
127    pub fn action(&mut self) -> Action<'_> {
128        self.insert(Name(b"A")).start()
129    }
130
131    /// Write the `/H` attribute to set what effect is used to convey that the
132    /// user is pressing a link annotation. Only permissible for the subtype
133    /// `Link`. PDF 1.2+.
134    pub fn highlight(&mut self, effect: HighlightEffect) -> &mut Self {
135        self.pair(Name(b"H"), effect.to_name());
136        self
137    }
138
139    /// Write the `/T` attribute. This is in the title bar of markup annotations
140    /// and should be the name of the annotation author. PDF 1.1+.
141    pub fn author(&mut self, text: TextStr) -> &mut Self {
142        self.pair(Name(b"T"), text);
143        self
144    }
145
146    /// Write the `/Subj` attribute. This is the subject of the annotation.
147    /// PDF 1.5+.
148    pub fn subject(&mut self, text: TextStr) -> &mut Self {
149        self.pair(Name(b"Subj"), text);
150        self
151    }
152
153    /// Write the `/QuadPoints` attribute, specifying the region in which the
154    /// link should be activated. PDF 1.6+.
155    pub fn quad_points(
156        &mut self,
157        coordinates: impl IntoIterator<Item = f32>,
158    ) -> &mut Self {
159        self.insert(Name(b"QuadPoints")).array().items(coordinates);
160        self
161    }
162
163    /// Write the `/L` attribute. This defines the start and end point of a
164    /// line annotation
165    pub fn line_to(&mut self, x1: f32, y1: f32, x2: f32, y2: f32) -> &mut Self {
166        self.insert(Name(b"L")).array().items([x1, y1, x2, y2]);
167        self
168    }
169
170    /// Start writing the `/FS` attribute, setting which file to reference.
171    pub fn file_spec(&mut self) -> FileSpec<'_> {
172        self.insert(Name(b"FS")).start()
173    }
174
175    /// Write the `/Name` attribute. Refer to the specification to see which
176    /// names are allowed for which annotation types.
177    pub fn icon(&mut self, icon: AnnotationIcon) -> &mut Self {
178        self.pair(Name(b"Name"), icon.to_name());
179        self
180    }
181}

185/// Kind of the annotation to produce.
186#[derive(Debug, Copy, Clone, Eq, PartialEq, Hash)]
187pub enum AnnotationType {
188    /// Inline text.
189    Text,
190    /// A link.
191    Link,
192    /// A line. PDF 1.3+.
193    Line,
194    /// A square. PDF 1.3+.
195    Square,
196    /// A circle. PDF 1.3+.
197    Circle,
198    /// Highlighting the text on the page. PDF 1.3+.
199    Highlight,
200    /// Underline the text on the page. PDF 1.3+.
201    Underline,
202    /// Squiggly underline of the text on the page. PDF 1.4+.
203    Squiggly,
204    /// Strike out the text on the page. PDF 1.3+.
205    StrikeOut,
206    /// A reference to another file. PDF 1.3+.
207    FileAttachment,
208}
209
210impl AnnotationType {
211    pub(crate) fn to_name(self) -> Name<'static> {
212        match self {
213            Self::Text => Name(b"Text"),
214            Self::Link => Name(b"Link"),
215            Self::Line => Name(b"Line"),
216            Self::Square => Name(b"Square"),
217            Self::Circle => Name(b"Circle"),
218            Self::Highlight => Name(b"Highlight"),
219            Self::Underline => Name(b"Underline"),
220            Self::Squiggly => Name(b"Squiggly"),
221            Self::StrikeOut => Name(b"StrikeOut"),
222            Self::FileAttachment => Name(b"FileAttachment"),
223        }
224    }
225}

275bitflags::bitflags! {
276    /// Bitflags describing various characteristics of annotations.
277    pub struct AnnotationFlags: u32 {
278        /// This will hide the annotation if the viewer does not recognize its
279        /// subtype. Otherwise, it will be rendered as specified in its appearance
280        /// stream.
281        const INVISIBLE = 1 << 0;
282        /// This hides the annotation from view and disallows interaction. PDF 1.2+.
283        const HIDDEN = 1 << 1;
284        /// Print the annotation. If not set, it will be always hidden on print.
285        /// PDF 1.2+.
286        const PRINT = 1 << 2;
287        /// Do not zoom the annotation appearance if the document is zoomed in.
288        /// PDF 1.3+.
289        const NO_ZOOM = 1 << 3;
290        /// Do not rotate the annotation appearance if the document is zoomed in.
291        /// PDF 1.3+.
292        const NO_ROTATE = 1 << 4;
293        /// Do not view the annotation on screen. It may still show on print.
294        /// PDF 1.3+.
295        const NO_VIEW = 1 << 5;
296        /// Do not allow interactions. PDF 1.3+.
297        const READ_ONLY = 1 << 6;
298        /// Do not allow the user to delete or reposition the annotation. Contents
299        /// may still be changed. PDF 1.4+.
300        const LOCKED = 1 << 7;
301        /// Invert the interpretation of the `no_view` flag for certain events.
302        /// PDF 1.5+.
303        const TOGGLE_NO_VIEW = 1 << 8;
304        /// Do not allow content changes. PDF 1.7+.
305        const LOCKED_CONTENTS = 1 << 9;
306    }
307}

322impl<'a> Action<'a> {
323    /// Write the `/S` attribute to set the action type.

325        self.pair(Name(b"S"), kind.to_name());
326        self
327    }
328
329    /// Start writing the `/D` attribute to set the destination of this

338        self.pair(Name(b"D"), name);
339        self
340    }
341
342    /// Start writing the `/F` attribute, setting which file to go to or which

346    }
347
348    /// Write the `/NewWindow` attribute to set whether this remote GoTo action
